Chapter 3 Generate Spring dimensionality reduction

3.1 Export data to Spring

dir.create(paste0(path_output, "springCoordinates"))

# Export raw expression matrix and gene list to generate a spring plot
exprData <- Matrix(as.matrix(data_seurat@assays$RNA@counts), sparse = TRUE)
writeMM(exprData, paste0(path_output, "springCoordinates/exprData.mtx"))

# Export gene list
geneList <- row.names(exprData)
write.table(geneList,
            paste0(path_output, "springCoordinates/geneList.csv"),
            sep="\t",
            col.names = F,
            row.names = F,
            quote = F)

# Export categorical metadata
cellMetadata  <- c('timepoint',
                   'replicate_id',
                   'individual',
                   'gender',
                   'Phase',
                   'Type_step1',
                   'Type_step2',
                   'Type_step2_unique',
                   'DV',
                   'Neuron_subtypes')
cellGroups <- do.call(rbind, lapply(data_seurat@meta.data[, cellMetadata], as.character))
write.table(cellGroups, paste0(path_output,"springCoordinates/cellGroups.csv"),
            sep =",",
            quote = FALSE,
            row.names = TRUE,
            col.names = FALSE)

# Export numerical metadata
cellMetadata  <- c('total_features',
                   'total_counts',
                   'pct_counts_MT',
                   'pct_counts_RP',
                   'Pseudotime',
                   'Progenitors_1',
                   'Neurons_1')
cellGroups <- do.call(rbind, data_seurat@meta.data[, cellMetadata])
getNArow <- unlist(lapply(cellMetadata, function(x) {any(is.na(cellGroups[x, ]))}))
getNAcol <- is.na(cellGroups[getNArow,])
cellGroups[getNArow, getNAcol] <- -10
write.table(cellGroups, paste0(path_output,"springCoordinates/colorTrack.csv"),
            sep =",",
            quote = FALSE,
            row.names = TRUE,
            col.names = FALSE)

# Export UMAP coordinates
umapCoord <- data_seurat@reductions$umap@cell.embeddings
rownames(umapCoord) <- NULL
write.table(umapCoord, paste0(path_output,"springCoordinates/customCoordinates.csv"),
            sep =",",
            quote = FALSE,
            row.names = FALSE,
            col.names = FALSE)

# Export gene sets
gskbDb <- c('mm_GO', 'mm_metabolic', 'mm_pathway', 'mm_TF', 'mm_location', 'mm_miRNA', 'mm_other')
data(list = gskbDb)
for (i in seq(1, length(gskbDb))){
    geneSets <- lapply(get(gskbDb[i]), function(x) x[-c(1,2)])
    geneSets <- do.call(qpcR:::rbind.na, geneSets)
    write.table(geneSets, paste0(path_output,"springCoordinates/geneSets_", gskbDb[i],".csv"),
                sep =",",
                na = "",
                quote = FALSE,
                row.names = TRUE,
                col.names = FALSE)
}
cd output/springCoordinates
# Remove trailing commas in geneSets.csv
sed -i 's/,*$//g' geneSets_*.csv
# Build short geneSets GO annotation focused on cell division
grep -e 'SYMMETRIC' -e 'DIVISION' geneSets_mm_GO.csv > geneSets_mm_GO_div.csv
