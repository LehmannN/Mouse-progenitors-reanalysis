4.1 k = 3

4.1.1 With TUBB3

set.seed(12345)
kident <- paste0('kmeanClust_Tubb3_', kcenter)
col.pal <- few_pal("Dark")(kcenter)
cl <- kmeans(cbind(dat.seurat@meta.data$Progenitors_1,
    dat.seurat@meta.data$Neurons_Tubb3_1),
    centers = kcenter)

dat.seurat@meta.data[[kident]] <- paste0("Clust.",cl$cluster)
table(dat.seurat@meta.data[[kident]])
## 
## Clust.1 Clust.2 Clust.3 
##   11787    8376    5411
prop.table(table(dat.seurat@meta.data[[kident]]))
## 
##   Clust.1   Clust.2   Clust.3 
## 0.4608978 0.3275201 0.2115821
p1 <- ggplot(dat.seurat@meta.data, aes(x = Progenitors_1, y = Neurons_Tubb3_1, alpha = 0.5)) +
    geom_point(aes_string(colour = kident)) +
    theme(legend.position="none") +
    scale_color_manual(values=col.pal)
ggMarginal(p1, type = "histogram", fill="lightgrey")

DimPlot(dat.seurat,
        group.by = kident,
        reduction = "umap",
        cols = col.pal,
        label.size = 4)

for(k in seq(1, kcenter, 1)){
    print(DimPlot(dat.seurat,
        group.by = kident,
        cells = which(dat.seurat@meta.data[[kident]] == paste0('Clust.', k)),
        reduction = "umap",
        cols = col.pal[k],
        label.size = 4))    
}
